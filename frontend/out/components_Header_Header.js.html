/**
 * My customization of the Navbar from MaterialUI. Responsible for triggering a change of page.
 */
function NavbarC(props) {
    const materialClasses = useStyles();
    const [drawer, setDrawer] = useState(false);

    let items = [{ link: '/', label: 'Submit' }, { link: '/admin', label: 'Admin'}, {link: '/references', label: 'References'}];

    const handleSearch = (char) => {
        props.search(char);
    };

    const handleClick = event => {
        props.history.push(event);
    };

    const sideList = () => (
        <div
            className={materialClasses.list}
            role='presentation'
            onClick={() => setDrawer(false)}
        >
            <List >
                {items.map(text => (
                    <ListItem
                        onClick={() => handleClick(text.link)}
                        button
                        key={text.label}
                    >
                        <ListItemText primary={text.label} />
                    </ListItem>
                ))}
            </List>
        </div>
    );

    return (
        <Fragment>
            <Drawer open={drawer} onClose={() => setDrawer(false)}>
                {sideList()}
            </Drawer>
            <AppBar className={materialClasses.appBar} position='static'>
                <Toolbar >
                    <IconButton
                        edge='start'
                        className={materialClasses.menuButton}
                        color='inherit'
                        aria-label='open drawer'
                        onClick={() => setDrawer(true)}
                    >
                        <MenuIcon />
                    </IconButton>
                    <Typography
                        className={materialClasses.title}
                        variant='h6'
                        noWrap
                    >
                        Radio Top 5
                    </Typography>
                </Toolbar>
            </AppBar>
        </Fragment>
    );
}


export default NavbarC;
